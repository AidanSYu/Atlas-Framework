#
#   pip install torch==2.5.1 --index-url https://download.pytorch.org/whl/cu121
#   pip install llama-cpp-python==0.3.19 --extra-index-url https://abetlen.github.io/llama-cpp-python/whl/cu121
#       # If no prebuilt CUDA wheel exists for the version, build from source
#       # (needs CUDA toolkit + MSVC):
#       #   $env:CMAKE_ARGS="-DGGML_CUDA=on"; pip install llama-cpp-python==0.3.19 --no-cache-dir
#   pip install -r requirements.txt
#
# `scripts/setup/setup_project.ps1` automates all of this on Windows.
